mình đã chỉnh sửa lại bảng inventorytransactions để phù hợp:
create table public.inventorytransactions (
  transaction_id bigint generated by default as identity not null,
  product_id bigint null,
  invoice_id text null,
  date date not null,
  amount integer null,
  type text null,
  note text null,
  source_type text null,
  constraint inventorytransactions_pkey primary key (transaction_id),
  constraint inventorytransactions_invoice_id_fkey foreign KEY (invoice_id) references invoices (invoice_id) on delete CASCADE
) TABLESPACE pg_default;
